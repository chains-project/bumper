static java.util.List<java.lang.String> getClassesLoc(java.io.File testSrcFolder) {
    SurefireBooterClient surefireBooterClient = new SurefireBooterClient();
    ForkConfiguration forkConfiguration = new ForkConfiguration(ForkMode.FORK_PER_CLASS);
    StartupConfiguration startupConfiguration = new StartupConfiguration();
    Commandline commandline = surefireBooterClient.createCommandLine(startupConfiguration, 1, testSrcFolder);
    ScannerFilter scannerFilter = new ScannerFilter();
    TestSetFilter testSetFilter = new JUnit4TestSetFilter();
    TestSetScanner testSetScanner = new JUnit4TestSetScanner(scannerFilter, testSetFilter);
    TestSetScannerContext testSetScannerContext = new JUnit4TestSetScannerContext();
    TestSetScannerResult testSetScannerResult = testSetScanner.scan(testSrcFolder, testSetScannerContext);
    ScanResult scanResult = new DefaultScanResult(testSetScannerResult);
    return scanResult.getClasses();
}
