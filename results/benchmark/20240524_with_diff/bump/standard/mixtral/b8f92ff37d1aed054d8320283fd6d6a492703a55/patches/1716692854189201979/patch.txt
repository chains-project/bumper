static java.util.List<java.lang.String> getClassesLoc(java.io.File testSrcFolder) {
    DefaultDirectoryScanner directoryScanner = new DefaultDirectoryScanner(testSrcFolder, new TestListResolver().getWildcard());
    return directoryScanner.scan().getClasses();
}

class DefaultDirectoryScanner extends DirectoryScanner {
    public DefaultDirectoryScanner(File baseDir, String includes) {
        super(baseDir, includes, Collections.<String>emptyList(), Collections.<String>emptyList(), false, false, false, false, false, false, false);
    }
}

class TestListResolver {
    public String getWildcard() {
        return "**/*.class";
    }
