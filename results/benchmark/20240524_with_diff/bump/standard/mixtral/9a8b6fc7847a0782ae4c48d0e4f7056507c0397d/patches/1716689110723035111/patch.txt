@java.lang.Override
protected org.jasypt.spring.security.PasswordEncoder createStringEncoder() {
    org.jasypt.spring.security.PasswordEncoder encoder = new org.jasypt.spring.security.PasswordEncoder();
    org.jasypt.util.password.PasswordEncryptor passwordEncryptor = new org.jasypt.util.password.StrongPasswordEncryptor();
    org.jasypt.util.text.TextEncryptor textEncryptor = new org.jasypt.util.text.StrongTextEncryptor();
    textEncryptor.setPassword(passwordEncryptor.getPassword());
    encoder.setTextEncryptor(textEncryptor);
    return encoder;
}
