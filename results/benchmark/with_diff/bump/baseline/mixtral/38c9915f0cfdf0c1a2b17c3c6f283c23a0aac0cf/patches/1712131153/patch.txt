import org.cactoos.text.Texts;

//...

@Override
public String asString(Iterable<WalletDto> wallets) throws IOException {
  return new JoinedText(Texts.quote("["),
    wallets::stream,
    w -> new JoinedText(",",
      Texts.quote("{"),
      w1 -> new JoinedText(",",
        Texts.of("id:", w1.id()),
        Texts.of("balance:", w1.balance())
      ),
      Texts.quote("}")
    ),
    Texts.quote("]")
  ).asString();
}
