import java.io.File;
import java.nio.file.Files;
import java.nio.file.Paths;

//...

private void deleteDirectory(File directory) throws IOException {
    if (directory.exists()) {
        Files.walk(Paths.get(directory.getAbsolutePath()))
            .sorted(Comparator.reverseOrder())
            .map(Path::toFile)
            .forEach(File::delete);
    }
}
