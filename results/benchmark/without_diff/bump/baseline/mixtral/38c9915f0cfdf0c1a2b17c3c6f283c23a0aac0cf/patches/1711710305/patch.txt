import org.cactoos.text.TextCollector;
import org.cactoos.text.Joined;

// ...

@Override
public TextWallets wallets(TextCursor cursor) throws IOException {
  return new Joined(
    "; ",
    new TextCollector(
      cursor.stream()
        .map(CursorIn::new)
        .toArray(CursorIn[]::new)
    )
  );
}
