static java.util.List<java.lang.String> getClassesLoc(java.io.File testSrcFolder) {
    org.apache.maven.plugin.surefire.util.DirectoryScanner directoryScanner = new org.apache.maven.plugin.surefire.util.DirectoryScanner(testSrcFolder, org.apache.maven.plugin.surefire.booterclient.lazytestprovider.Commandline.getWildcard());
    return directoryScanner.scan().getClasses();
}
