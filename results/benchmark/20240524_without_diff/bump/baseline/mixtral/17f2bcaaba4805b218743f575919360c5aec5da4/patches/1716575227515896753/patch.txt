import processing.core.PShape;

import static org.junit.Assert.assertEquals;

@Test
void test_PGS_Morphology_smooth() {
    assumeTrue(GROUP_SHAPE.getChildCount() == 2);
    PShape pgsMorphology = new PGS_Morphology();
