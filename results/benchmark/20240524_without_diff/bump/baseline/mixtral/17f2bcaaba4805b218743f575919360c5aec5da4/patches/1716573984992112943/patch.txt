@Test
void test_PGS_Morphology_minkSum() {
    assumeTrue(GROUP_SHAPE.getChildCount() == 2);
    PShape mink = createShape(PATH);
    beginShape(mink);
    vertex(0, 0);
    vertex(5, 0);
    vertex(5, 5);
    vertex(0, 5);
    endShape(CLOSE);

    PShape out = PGS_Morphology.minkSum(GROUP_SHAPE, mink);
    assertEquals(2, out.getChildCount());
}
