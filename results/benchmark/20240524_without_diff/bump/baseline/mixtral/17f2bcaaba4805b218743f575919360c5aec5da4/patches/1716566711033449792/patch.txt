import processing.core.PShape;

@Test
void test_PGS_Morphology_fieldWarp() {
    assumeTrue(GROUP_SHAPE != null && GROUP_SHAPE.getChildCount() == 2);
    PShape processedGroupShape = GROUP_SHAPE.getChildren()[0];
