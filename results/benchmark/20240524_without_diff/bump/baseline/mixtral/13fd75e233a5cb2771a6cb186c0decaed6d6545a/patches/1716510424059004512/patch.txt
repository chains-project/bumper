@Test
void largeImageUploadWorks() throws Exception {

        this.buildImage();
        final String output = this.client.run("push", this.remote());
        MatcherAssert.assertThat(output, new StringContains(pushed ? "Pushed" : ""));
    } finally {
        this.client.run("rmi", this.remote());
    }
}
